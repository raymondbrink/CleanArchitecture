public static class ProjectPhaseStatusHelper
{
        public static Expression<Func<ProjectPhase, ProjectPhaseStatusModel>> Status =>
            phase =>
                HasBuyersAttendants.Invoke(phase)
                    ? ProjectPhaseHasBuyer.Invoke(phase)
                        ? HasBuyerWithinOrderTime.Invoke(phase)
                            ? AllObjectsHavePassedClosingDate.Invoke(phase)
                                ? ProjectPhaseStatusModel.Closed
                                : ProjectPhaseStatusModel.Buying
                            : ProjectPhaseStatusModel.Orientation
                        : ProjectPhaseStatusModel.WaitingForBuyers            
                    : ProjectPhaseStatusModel.New;
}

public enum ProjectPhaseStatusModel
{
    New = 0,
    WaitingForBuyers = 1,
    Orientation = 2,
    Buying = 3,
    Closed = 4
}

CreateMap<ProjectPhase, ProjectPhaseBaseModel>()
    .ForMember(d => d.Status, a => a.MapFrom(s => ProjectPhaseStatusHelper.Status.Invoke(s)));
